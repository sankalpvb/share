using System;
using System.Data;
using System.Xml;

namespace Praactical_Exam
{
    public partial class XmlDemo : System.Web.UI.Page
    {
        string path;

        protected void Page_Init(object sender, EventArgs e)
        {
            // Path to App_Data folder
            path = Server.MapPath("~/App_Data/student.xml");

            // Create XML file if not exists
            if (!System.IO.File.Exists(path))
            {
                XmlDocument doc = new XmlDocument();
                XmlElement root = doc.CreateElement("Students");
                doc.AppendChild(root);
                doc.Save(path);
            }
        }

        protected void btnWrite_Click(object sender, EventArgs e)
        {
            XmlDocument doc = new XmlDocument();
            doc.Load(path);

            XmlElement student = doc.CreateElement("Student");

            XmlElement name = doc.CreateElement("Name");
            name.InnerText = txtName.Text;

            XmlElement age = doc.CreateElement("Age");
            age.InnerText = txtAge.Text;

            student.AppendChild(name);
            student.AppendChild(age);

            doc.DocumentElement.AppendChild(student);
            doc.Save(path);

            lblOutput.Text = "Data written successfully!";
        }

        protected void btnRead_Click(object sender, EventArgs e)
        {
            DataSet ds = new DataSet();
            ds.ReadXml(path);

            lblOutput.Text = "<b>Students:</b><br/>";

            foreach (DataRow row in ds.Tables[0].Rows)
            {
                lblOutput.Text += row["Name"] + " - " + row["Age"] + "<br/>";
            }
        }
    }
}
